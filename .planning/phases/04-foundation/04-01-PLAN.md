  <name>Task 2: Create getPublishedPosts helper function</name>
  <files>src/utils/posts.ts</files>
  <action>
    创建 `src/utils/posts.ts` 文件，导出 `getPublishedPosts()` 函数。

    函数实现逻辑:
    ```typescript
    import { getCollection } from 'astro:content';

    /**
     * 获取已发布文章列表
     * - 生产环境 (import.meta.env.PROD): 返回非草稿文章 (draft !== true)
     * - 开发环境: 返回所有文章 (便于预览草稿)
     */
    export async function getPublishedPosts() {
      const posts = await getCollection('posts');

      if (import.meta.env.PROD) {
        return posts.filter((post) => post.data.draft !== true);
      }

      return posts;
    }
    ```

    使用 `import.meta.env.PROD` 区分构建环境:
    - Vercel 生产构建时为 true，过滤掉草稿
    - 本地 `npm run dev` 时为 false，显示所有文章
  </action>
  <verify>
    <automated>ls src/utils/posts.ts && head -5 src/utils/posts.ts</automated>
  </verify>
  <done>src/utils/posts.ts 存在，导出 getPublishedPosts 函数</done>
